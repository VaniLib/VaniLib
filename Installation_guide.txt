# To install conda and create the environment, run the following commands
1. wget https://repo.anaconda.com/miniconda/Miniconda3-latest-Linux-x86_64.sh
2. bash Miniconda3-latest-Linux-x86_64.sh
3. conda env create -n iitm -f ./requirements.yaml

# to first debug the code, run the following command (RECOMMENDED)
python main.py

# install redis server on your machine by following the instructions in the following link
1. https://redis.io/docs/latest/operate/oss_and_stack/install/install-redis/install-redis-on-linux/

# to start the celery server and redis for background tasks, run the following command
1. redis-server
2. celery -A setup_server:celery_app worker -B --loglevel INFO
3. python main.py
